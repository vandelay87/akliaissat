    <div class="block">

        <Heading :title="social.name" align="centre" :size=2  />
        <Button :value="social.account" :onClick="(event) => { link(event, index) }" :customClass="social.name" />
      </section>
    </div>
  </article>
</template>
 
<script>
  import Heading from './generic/Heading'
  import Button from './generic/Button'

  }
</script>
